51/*! \brief retrieve an R/G/B/Alpha (\a ch == 0/1/2/3) value from the \c QRgb value \a rgb
52 \ingroup jkqtplotter_imagelots_tools
53 */
54inline int JKQTPGetColorChannel(QRgb rgb, int ch)
55{
56 switch(ch) {
57 case 0: return qRed(rgb);
58 case 1: return qGreen(rgb);
59 case 2: return qBlue(rgb);
60 case 3: return qAlpha(rgb);
61 }
62 return qGray(rgb);
63}
64
65/*! \brief set the R/G/B/Alpha (\a ch == 0/1/2/3) value in the \c QRgb value \a rgb to \a val (0..255!)
66 \ingroup jkqtplotter_imagelots_tools
67 */
68inline void JKQTPSetColorChannel(QRgb& rgb, int ch, int val)
69{
70 switch(ch) {
71 case 0: rgb= qRgba(val, qGreen(rgb), qBlue(rgb), qAlpha(rgb)); break;
72 case 1: rgb= qRgba(qRed(rgb), val, qBlue(rgb), qAlpha(rgb)); break;
73 case 2: rgb= qRgba(qRed(rgb), qGreen(rgb), val, qAlpha(rgb)); break;
74 case 3: rgb= qRgba(qRed(rgb), qGreen(rgb), qBlue(rgb), val); break;
75 }
76}

292 /*! \brief convert a 2D image (as 1D array) into a QImage with given palette (see JKQTFPColorPalette)
293
294 \param dbl_in pointer to a 1D array of template type \c T representing the image to plot. This array has to be of size \a width * \a height
295 \param width width of the array in \a dbl
296 \param height height of the array in \a dbl
297 \param[out] img the QImage object to draw to (should be initialized as \c QImage::Format_ARGB32 )
298 \param palette the color palette to use for the display
299 \param minColor lower boundary of color range in \a dbl pixels, if \a minColor == \a maxColor then this function will extract the image min and image max.
300 \param maxColor upper boundary of color range in \a dbl pixels, if \a minColor == \a maxColor then this function will extract the image min and image max.
301 \param paletteMinFail specifies what shell happen, when a value in \a dbl is below \a minColor
302 \param paletteMaxFail specifies what shell happen, when a value in \a dbl is above \a maxColor
303 \param minFailColor color to use for pixels that are below \a minColor for some settings of \a paletteMinFail
304 \param maxFailColor color to use for pixels that are below \a maxColor for some settings of \a paletteMaxFail
305 \param nanColor color to use for pixels that are not-a-number
306 \param infColor color to use for pixels that are infinity
307 \param logScale create a log-scaled image
308 \param logBase base for the logarithm used when \c logScale==true
309 \param lutUser user define LUT, used if \a palette \c ==JKQTPMathImageUSER_PALETTE
310 \note There is a variant of this function that is called with a userLUT directly, instead of \a palette
311 */
312 template <class T>
313 static inline void array2image(const T* dbl_in, int width, int height, QImage &img, JKQTPMathImageColorPalette palette, double minColor, double maxColor, JKQTPMathImageColorRangeFailAction paletteMinFail=JKQTPMathImageLastPaletteColor, JKQTPMathImageColorRangeFailAction paletteMaxFail=JKQTPMathImageLastPaletteColor, QColor minFailColor=QColor("black"), QColor maxFailColor=QColor("black"), QColor nanColor=QColor("black"), QColor infColor=QColor("black"), bool logScale=false, double logBase=10.0, const LUTType& lutUser=LUTType())
314 {
315 if (!dbl_in || width<=0 || height<=0)
316 return;
317
318 double min = *dbl_in;
319 double max = *dbl_in;
320 if (jkqtp_approximatelyEqual(minColor, maxColor, JKQTP_DOUBLE_EPSILON)) {
321 bool first=true;
322 for (int i=1; i<width*height; ++i)
323 {
324 T v=dbl_in[i];
325 if (!(std::isnan(static_cast<long double>(v)) || std::isinf(static_cast<long double>(v)))) {
326 if (first) {
327 min=max=v;
328 first=false;
329 } else {
330 if (v < min)
331 min = v;
332 else if (v > max)
333 max = v;
334 }
335 }
336 }
337
338 } else {
339 min = minColor;
340 max = maxColor;
341 }
342
343 const T* dbl=dbl_in;
344 QVector<T> dbl1;
345 if (logScale) {
346 double logB=log10(logBase);
347 dbl1=QVector<T>(width*height, 0);
348 for (int i=0; i<width*height; i++) {
349 dbl1[i]=log10(dbl_in[i])/logB;
350 }
351 dbl=dbl1.data();
352 min=log10(min)/logB;
353 max=log10(max)/logB;
354 }
355 double delta=max-min;
356
357
358 // either use lutUser (if palette==JKQTPMathImageUSER_PALETTE) or else read the LUT from the internal storage
359 const LUTType& lut_used=(palette==JKQTPMathImageUSER_PALETTE)?lutUser:getLUTforPalette(JKQTPImageTools::global_jkqtpimagetools_lutstore, palette);
360
361 img = QImage(width, height, QImage::Format_ARGB32);
362 if (jkqtp_approximatelyEqual(min, max, JKQTP_DOUBLE_EPSILON)) {
363 img.fill(0);
364 } else {
365 const int lutSize=lut_used.size();
366 if (lutSize>0) {
367 // LUT found: collor the image accordingly
368 for (int j=0; j<height; ++j) {
369 QRgb* line=reinterpret_cast<QRgb*>(img.scanLine(height-1-j));
370 for (int i=0; i<width; ++i) {
371 double val=dbl[j*width+i];
372 if (std::isnan(val)) {
373 line[i]=nanColor.rgba();
374 } else if (std::isinf(val)) {
375 line[i]=infColor.rgba();
376 } else {
377 const int v = static_cast<int>((val-min)/delta*static_cast<double>(lutSize-1));
378 const int vv = (v < 0) ? 0 : ( (v >= lutSize) ? (lutSize-1) : v);
379 line[i]=lut_used[vv];
380 if ((v<0)&&(paletteMinFail==JKQTPMathImageGivenColor)) {
381 line[i]=minFailColor.rgba();
382 } else if ((v>lutSize)&&(paletteMaxFail==JKQTPMathImageGivenColor)) {
383 line[i]=maxFailColor.rgba();
384 } else if ((v<0)&&(paletteMinFail==JKQTPMathImageTransparent)) {
385 line[i]=QColor(Qt::transparent).rgba();
386 } else if ((v>lutSize)&&(paletteMaxFail==JKQTPMathImageTransparent)) {
387 line[i]=QColor(Qt::transparent).rgba();
388 }
389 }
390 }
391 }
392 } else {
393 // no LUT found: paint a black image!
394 img.fill(0);
395 }
396 }
